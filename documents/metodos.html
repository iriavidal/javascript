Se utiliza para crear y devolver una nueva cadena que consiste en la repetición de una cadena original un número determinado de veces. 

const mensaje = "Hola ";

// Utilizando repeat para repetir la cadena "Hola " tres veces
const saludoRepetido = mensaje.repeat(3);
            
console.log(saludoRepetido);
// Salida: "Hola Hola Hola "

Se utiliza para extraer una porción (subcadena) de una cadena o array, sin modificar la cadena o array original.

Para cadenas:

const mensaje = "Hola, mundo";

// Utilizando slice para obtener una porción de la cadena
const porcion = mensaje.slice(0, 4);

console.log(porcion);
// Salida: "Hola"

Para arrays:

const numeros = [1, 2, 3, 4, 5];

// Utilizando slice para obtener una porción del array
const porcionArray = numeros.slice(1, 4);
    
console.log(porcionArray);
// Salida: [2, 3, 4]
// Función que verifica si un número es automorfo
function automorfo(n) {
  // Calcula el cuadrado de n y lo convierte a cadena
  let n2 = Math.pow(n, 2).toString();
            
  // Obtiene los últimos dígitos del cuadrado, donde dígitos es la cantidad de dígitos en n
  let final = Array.from(n2).slice(-n.length);
            
  // Compara el número original con los últimos dígitos del cuadrado
  return n == final.join("");
}

let n;
            
// Ciclo do-while para solicitar un número válido al usuario
do {
  n = prompt("Número");
} while (isNaN(n) || n < 1);
            
// Llama a la función automorfo y muestra el resultado en la consola
console.log(automorfo(n));

Se utiliza con cadenas de texto. La función de endsWith es verificar si una cadena termina con otra cadena específica. Devuelve true si la cadena original termina con la cadena proporcionada y false en caso contrario. 

let frase = "Hola, mundo";

// Verifica si la frase termina con "mundo"
let terminaConMundo = frase.endsWith("mundo");
            
console.log(terminaConMundo); // Devolverá true
function automorfo2(n) {
  return Math.pow(n, 2).toString().endsWith(n);

Tipos de funciones

Apuntes Javascript 
// Función tradicional
function sumar(a, b) {
  return a + b;
}

// Función de flecha equivalente
const sumar = (a, b) => a + b;
// Sintaxis de expresión de función anónima
const miFuncion = function(parametro1, parametro2) {
  // Cuerpo de la función
  // Puedes hacer algo con los parámetros aquí
  console.log(parametro1 + parametro2);
};
            
// Llamada a la función anónima
miFuncion(3, 4); // Salida: 7
